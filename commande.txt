ls = lister les fichiers/dossier
cd = changer de repetoire
cd ./.. = pour remonter 2 fois
cd .. = pour remonter une fois
clear = pour nettoyer la fenetre terminal
mkdir *nom du dossier = créer un dossier
touch *nom du fichier = créer un fichier
nano exo2.txt = créer un fichier et ouvrir pour écrire
cat *nom du fichier = voir le contenu du fichier
ls -l = les droits des fichiers
*(nom de la commande) --help = donne des informations sur les commandes
find (début du terme rechercher "sensible à la casse")* -print
grep "termes rechercher" nom du fichier où chercher = cherche dans le fichier le terme dans la ligne
rm = supprimer un fichier
rm -r = supprimer un dossier
rmdir = supprimer un dossier aussi
-zcvf = compresser
mv source destination = deplace un fichier
